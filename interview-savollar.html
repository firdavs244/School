def my_decorator(func):
    def wrapper(*args, **kwargs):
        print("Funksiyadan oldin")
        result = func(*args, **kwargs)
        print("Funksiyadan keyin")
        return result
    return wrapper

def say_hello():

17JWT nima?O'rta+

JWT - JSON Web Token. Foydalanuvchini aniqlash uchun token.

Tuzilishi (3 qism):

  • Header: algoritm va token turi
  • Payload: ma'lumotlar (user_id, email, exp)
  • Signature: imzo (secret key bilan)

eyJzdWIiOiJhbGlAbWFpbC5jb20ifQ.  # Payload
SflKxwRJSMeKKF2QT4fwpMeJf36POk  # Signature

from passlib.context import CryptContext

pwd_context = CryptContext(schemes=["bcrypt"])

def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)

23HTTP status kodlari nimalarni bildiradi?Oson+
  • 200 OK - muvaffaqiyatli
  • 201 Created - yangi resurs yaratildi
  • 400 Bad Request - noto'g'ri so'rov
  • 401 Unauthorized - autentifikatsiya kerak
  • 403 Forbidden - ruxsat yo'q
  • 404 Not Found - topilmadi
  • 500 Internal Error - server xatosi
24Path parameter va Query parameter farqi?Oson+
# Path parameter - URL ichida
# GET /users/5
@router.get("/users/{user_id}")
def get_user(user_id: int):
    return db.query(User).filter(User.id == user_id).first()

# Query parameter - URL oxirida ?key=value
# GET /users?role=student&limit=10
@router.get("/users")
def get_users(role: str = None, limit: int = 10):
    query = db.query(User)
    if role:
        query = query.filter(User.role == role)
    return query.limit(limit).all()
